\begin{simbolos}
  \item[$ \alpha $] Alfa ou Autovalor maior de \textit{keypoint}
  \item[$ \beta $] Beta ou Autovalor menor de \textit{keypoint}
  \item[$ \sigma $] Sigma ou desvio padrão
  \item[$ \theta $] Angulo
  \item[$ \partial $] Diferencial
  \item[$ \hat{x} $] Extrema
  \item[$ \pi $] Pi ou $180^{\circ}$
